Validación REAL contra la API de Lemon Squeezy async function validateLicense() { const key = prompt("Introduce tu clave de licencia de Lemon Squeezy:"); if (!key || key.trim() === "") return; // Mostramos el loader para que el usuario sepa que está cargando const loader = document.getElementById('loader'); document.getElementById('loadingText').innerText = "Validando licencia con Lemon Squeezy..."; loader.classList.remove('hidden'); try { // Hacemos la petición a la API pública de verificación de licencias const response = await fetch('https://api.lemonsqueezy.com/v1/licenses/validate', { method: 'POST', headers: { 'Accept': 'application/json', 'Content-Type': 'application/x-www-form-urlencoded' }, body: new URLSearchParams({ license_key: key.trim() }) }); const data = await response.json(); // Verificamos la respuesta oficial del servidor if (data.valid) { unlockApp(); localStorage.setItem('isProUser', 'true'); alert("¡Licencia validada con éxito! Disfruta de la versión Pro."); } else { // Si no es válida, mostramos el error (expirada, desactivada, no existe...) alert("La licencia no es válida. Motivo: " + (data.error || "Clave incorrecta")); } } catch (error) { console.error("Error validando licencia:", error); alert("Hubo un error al conectar con el servidor de licencias. Revisa tu conexión."); } finally { // Ocultamos el loader al terminar loader.classList.add('hidden'); } } function unlockApp() { document.getElementById('lockedState').classList.add('hidden'); document.getElementById('unlockedState').classList.remove('hidden'); } window.onload = function() { // Mantiene la app desbloqueada si ya se validó en una visita anterior if(localStorage.getItem('isProUser') === 'true') { unlockApp(); } if(window.LemonSqueezy) { window.LemonSqueezy.Setup({ eventHandler: window.lemonSqueezyUrl.eventHandler }); } };
